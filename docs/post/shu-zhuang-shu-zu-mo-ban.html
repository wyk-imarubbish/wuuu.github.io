树状数组模板

/**********************************************************
 * @name： 树状数组
 * @author: wyk
 * @function: 区间修改，区间查询
 * @note: 区间修改单点查询、单点修改区间查询可以直接用这个板子
 **********************************************************/

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

template <typename T>
class TreeArray {
   private:
    vector<T> a, b;
    int lowbit(int x) { return x & -x; }
    // 对差分数组的修改，不在外部调用
    void update(int pos, T val) {
        for (int i = pos; i < a.size(); i += lowbit(i)) {
            a[i] += val;
            b[i] += pos * val;
        }
    }

   public:
    TreeArray(int n) : a(n + 1), b(n + 1) {}
    // 区间修改
    void updateRange(int l, int r, T val) {
        update(l, val);
        update(r + 1, -val);
    }
    // 前缀和
    T prefixSum(int pos) {
        T as = 0, bs = 0;
        for (int i = pos; i > 0; i -= lowbit(i)) {
            as += a[i];
            bs += b[i];
        }
        return (pos + 1) * as - bs;
    }
    // 查询区间和
    T rangeSum(int l, int r) { return prefixSum(r) - prefixSum(l - 1); }
    // Θ(n) 建树
    void buildTree(vector<T>& arr) {
        for (int i = 1; i < arr.size(); i++) {
            update(i, arr[i] - arr[i - 1]);
        }
    }
};

// Luogu P3372
int main() {
    int n, m;
    cin >> n >> m;
    vector<ll> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    TreeArray<ll> tree_a(n);
    tree_a.buildTree(a);

    for (int i = 1; i <= m; i++) {
        int op, x, y, k;
        cin >> op >> x >> y;
        if (op == 1) {
            cin >> k;
            tree_a.updateRange(x, y, k);
        } else {
            cout << tree_a.rangeSum(x, y) << endl;
        }
    }
}
